\begin{center}
\begin{tikzpicture}[
    scale=0.6,
    >=Latex,
    every node/.style={font=\small}

\def\ymax{3}

% Axes
\draw[very thick, ->] (\xmin,0) -- (\xmax+0.2,0);
\draw[very thick, ->] (0,\ymin) -- (0,\ymax+0.2);

% Points
\coordinate (A) at (#1,#2);
\coordinate (B) at (#3,#4);

% Calcul pente
\pgfmathsetmacro{\a}{(#4-#2)/(#3-#1)}
\pgfmathsetmacro{\b}{#2-\a*#1}

% Tracé de la droite limité à la fenêtre
\draw[thick, domain=\xmin:\xmax] 
    plot (\x, {\a*\x+\b});

% Points A et B
\fill[blue] (A) circle (3pt);
\fill[blue] (B) circle (3pt);
\node[blue, below left] at (A) {$A$};
\node[blue, above right] at (B) {$B$};

% Origine
\node[below left] at (0,0) {$O$};

\end{tikzpicture}
\end{center}
}

\section{Équation d'une droite}

\subsection{Définition}
\defi Dans un repère, une droite peut s’écrire :
\[
\begin{cases}
x=c & \text{si la droite est parallèle à l'axe des ordonnées} \\
y=ax+b & \text{si la droite n'est pas parallèle à l'axe des ordonnées}
\end{cases}
\]

\prop Pour $y=ax+b$ :
\begin{itemize}
  \item $a$ est le \textbf{coefficient directeur} (pente)
  \item $b$ est l'\textbf{ordonnée à l'origine} : la droite passe par $(0\,;b)$
\end{itemize}

% -------- Graphique (exemple) --------
% Change les coordonnées pour ton cours :
% Ici : A(-2,-1) et B(3,1)
\GraphDroiteAB{-2}{2}{4}{-1}

\subsection{Déterminer $a$ et $b$ avec deux points}
Soient $A(x_A;y_A)$ et $B(x_B;y_B)$ avec $x_A\neq x_B$ :
\[
a=\frac{y_B-y_A}{x_B-x_A}
\qquad
b=y_A-ax_A
\]
\rem Si $x_A=x_B$, la droite est \textbf{verticale} : $x=x_A$.

\section{Variations}
Pour $f(x)=ax+b$ :
\begin{itemize}
  \item si $a>0$, $f$ est \textbf{croissante}
  \item si $a<0$, $f$ est \textbf{décroissante}
  \item si $a=0$, $f$ est \textbf{constante} : $f(x)=b$ (droite horizontale)
\end{itemize}

\section{Signe de $ax+b$ (inéquations)}
\subsection{Zéro de $ax+b$}
Si $a\neq 0$ :
\[
ax+b=0 \iff x=-\frac{b}{a}
\]

\section{À retenir}
\begin{itemize}
  \item Droite affine : $y=ax+b$
  \item $a=\dfrac{\Delta y}{\Delta x}$ ; $b=f(0)$.
  \item $ax+b=0 \iff x=-\dfrac{b}{a}$ (si $a\neq 0$).
\end{itemize}
