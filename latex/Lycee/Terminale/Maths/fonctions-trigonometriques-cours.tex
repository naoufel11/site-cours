\section{Fonction cosinus}

\subsection{Définition et propriétés}
$\cos x$ est l'abscisse d'un point $M$ du cercle trigonométrique.

\begin{itemize}
\item \textbf{Domaine de définition :} $\cos x$ est définie pour tout $x\in\mathbb{R}$
\item \textbf{propriété :} $-1\leq \cos x\leq 1$
\item \textbf{périodicité :} $\cos(x+2\pi)=\cos x$
\item \textbf{parité :} $\cos(-x)=\cos x$
\item \textbf{signe :} $\cos x\geq 0$ pour $x\in\left[-\frac{\pi}{2};\frac{\pi}{2}\right]$
\item \textbf{dérivée :} $(\cos x)'=-\sin x$
\item \textbf{limite :} $\displaystyle \lim_{x\to 0}\frac{\cos x-1}{x}=0$
\end{itemize}

\subsection{Tableau de variations}
Étude réduite à \ $x\in[0;\pi]$ \ car \ $\cos x$ \ fonction paire et $2\pi$-périodique

\VarCos

\subsection{Représentation graphique}
\GraphCos

\section{Fonction sinus}

\subsection{Définition et propriétés}
$\sin x$ est l'ordonnée d'un point $M$ du cercle trigonométrique.

\begin{itemize}
\item \textbf{domaine de définition :} $\sin x$ est définie pour tout $x\in\mathbb{R}$
\item \textbf{propriété :} $-1\leq \sin x\leq 1$
\item \textbf{périodicité :} $\sin(x+2\pi)=\sin x$
\item \textbf{parité :} $\sin(-x)=-\sin x$
\item \textbf{signe :} $\sin x\geq 0$ pour $x\in[0;\pi]$
\item \textbf{dérivée :} $(\sin x)'=\cos x$
\item \textbf{limite :} $\displaystyle \lim_{x\to 0}\frac{\sin x}{x}=1$
\end{itemize}

\subsection{Tableau de variations}
Étude réduite à \ $x\in[0;\pi]$ \ car \ $\sin x$ \ fonction impaire et $2\pi$-périodique

\VarSin

\subsection{Représentation graphique}
\GraphSin

\section{Résolution d'équations}

\subsection{Équation avec le cosinus}
\[
\cos(x)=\cos(a)
\Longleftrightarrow
\left\{
\begin{array}{l}
x=a+2k\pi\\[0.2em]
x=-a+2k\pi
\end{array}
\right.
\qquad ,\ k\in\mathbb{Z}
\]

\EqCosCircle

\subsection{Équation avec le sinus}
\[
\sin(x)=\sin(a)
\Longleftrightarrow
\left\{
\begin{array}{l}
x=a+2k\pi\\[0.2em]
x=\pi-a+2k\pi
\end{array}
\right.
\qquad ,\ k\in\mathbb{Z}
\]

\EqSinCircle
